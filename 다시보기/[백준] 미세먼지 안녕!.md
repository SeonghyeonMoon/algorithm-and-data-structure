# [백준] 미세먼지 안녕!

```js
const initiateGraph = (r) => {
  const result = [];
  for (let i = 0; i < r; i++) {
    result.push(input().split(' ').map(Number));
  }
  return result;
}

const findCleaner = (r, c, graph) => {
  for (let x = 0; x < r; x++) {
    for (let y = 0; y < c; y++) {
      if (graph[x][y] === -1) {
        return x;
      }
    }
  }
}

const findDust = (graph) => {
  const result = [];
  for (let x = 0; x < graph.length; x++) {
    for (let y = 0; y < graph[x].length; y++) {
      if (graph[x][y] > 4) result.push([x, y]);
    }
  }
  return result;
}

const move = [[-1, 0], [1, 0], [0, -1], [0, 1]];

const diffuseGraph = (r, c, graph) => {
  const dusts = findDust(graph);
  const result = graph.map((row) => [...row]);
  for (const [x, y] of dusts) {
    const diffused = Math.floor(graph[x][y] / 5);
    for (const [dx, dy] of move) {
      const [nx, ny] = [x + dx, y + dy];
      if (nx < 0 || nx >= r || ny < 0 || ny >= c) continue;
      if (graph[nx][ny] === -1) continue;
      result[nx][ny] += diffused;
      result[x][y] -= diffused;
    }
  }
  return result;
}

const circulateGraph = (r, c, cleaner, graph) => {
  const x1 = cleaner;
  for (let x = x1 - 1; x > 0; x--) graph[x][0] = graph[x - 1][0];
  for (let y = 0; y < c - 1; y++) graph[0][y] = graph[0][y + 1];
  for (let x = 0; x < x1; x++) graph[x][c - 1] = graph[x + 1][c - 1];
  for (let y = c - 1; y > 1; y--) graph[x1][y] = graph[x1][y - 1];
  graph[x1][1] = 0;
  const x2 = cleaner + 1;
  for (let x = x2 + 1; x < r - 1; x++) graph[x][0] = graph[x + 1][0];
  for (let y = 0; y < c; y++) graph[r - 1][y] = graph[r - 1][y + 1];
  for (let x = r - 1; x > x2; x--) graph[x][c - 1] = graph[x - 1][c - 1];
  for (let y = c - 1; y > 1; y--) graph[x2][y] = graph[x2][y - 1];
  graph[x2][1] = 0;
  return graph;
}

const countDust = (graph) => graph.reduce((acc, row) => acc + row.reduce((acc, col) => acc + col, 0), 2);

const solution = () => {
  const [r, c, t] = input().split(' ').map(Number);
  let graph = initiateGraph(r);
  const cleaner = findCleaner(r, c, graph);

  for (let i = 0; i < t; i++) {
    graph = diffuseGraph(r, c, graph);
    graph = circulateGraph(r, c, cleaner, graph);
  }

  return countDust(graph);
}

console.log(solution());
```
