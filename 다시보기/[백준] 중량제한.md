# [백준] [중량제한](https://www.acmicpc.net/problem/1939)

## 풀이 1

이진 탐색

```js
const [n, m] = input().split(' ').map(Number);
let max = 0;

const graph = {};
for (let i = 0; i < m; i++) {
  const [a, b, c] = input().split(' ').map(Number);
  max = Math.max(max, c);
  graph[a] = graph[a] || [];
  graph[a].push([b, c]);
  graph[b] = graph[b] || [];
  graph[b].push([a, c]);
}

const [s, e] = input().split(' ').map(Number);

const isCanMove = (cost) => {
  const visited = new Set();
  const queue = [s];
  while (queue.length) {
    const cur = queue.pop();
    if (cur === e) return true;
    visited.add(cur);
    for (const [next, c] of graph[cur]) {
      if (visited.has(next)) continue;
      if (c < cost) continue;
      queue.push(next);
    }
  }
  return false;
}

const binarySearch = (left, right) => {
  while (left <= right) {
    const mid = Math.floor((left + right) / 2);
    isCanMove(mid) ? left = mid + 1 : right = mid - 1;
  }
  return right;
}

console.log(binarySearch(0, max));
```

## 풀이 2

유니온 파인드

```js
const [n, m] = input().split(' ').map(Number);

const edges = [];
for (let i = 0; i < m; i++) {
  const [a, b, c] = input().split(' ').map(Number);
  edges.push([a, b, c]);
}
edges.sort((a, b) => b[2] - a[2]);

const find = (x) => {
  if (parent[x] === x) return x;
  return parent[x] = find(parent[x]);
}

const union = (a, b) => {
  a = find(a);
  b = find(b);
  if (a === b) return false;
  if (a > b) [a, b] = [b, a];
  parent[b] = a;
  return true;
}

const [s, e] = input().split(' ').map(Number);

const parent = Array(n + 1).fill(0).map((_, i) => i);
for (let i = 0; i < m; i++) {
  const [a, b, c] = edges[i];
  union(a, b);
  if (find(s) === find(e)) {
    console.log(c);
    break;
  }
}
```
