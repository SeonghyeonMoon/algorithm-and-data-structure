# [백준] 가장 긴 증가하는 부분 수열

```js
const solution = () => {
  const n = Number(input());
  const arr = input().split(' ').map(Number);

  const result = [arr[0]];

  const bisect = (arr, num, start, end) => {
    if (start === end) return start;

    const mid = Math.floor((start + end) / 2);

    if (arr[mid] < num) {
      return bisect(arr, num, mid + 1, end);
    } else {
      return bisect(arr, num, start, mid);
    }
  }

  for (let i = 1; i < n; i++) {
    const num = arr[i];
    if (result[result.length - 1] < num) {
      result.push(num);
      continue;
    }

    const index = bisect(result, num, 0, result.length - 1);
    result[index] = num;
  }

  return result.length;
}

console.log(solution());
```
