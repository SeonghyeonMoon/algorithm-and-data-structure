# [백준] 가장 긴 바이토닉 부분

```js
const solution = () => {
  const n = Number(input());
  const sequence = input().split(' ').map(Number);

  const ascending = Array(n).fill(1);
  for (let i = 1; i < n; i++) {
    for (let j = 0; j < i; j++) {
      if (sequence[i] > sequence[j]) {
        ascending[i] = Math.max(ascending[i], ascending[j] + 1);
      }
    }
  }

  const descending = Array(n).fill(1);
  for (let i = n - 2; i >= 0; i--) {
    for (let j = n - 1; j > i; j--) {
      if (sequence[i] > sequence[j]) {
        descending[i] = Math.max(descending[i], descending[j] + 1);
      }
    }
  }

  let result = 0;
  for (let i = 0; i < n; i++) {
    result = Math.max(result, ascending[i] + descending[i]);
  }
  return result - 1;
}

console.log(solution());
```
