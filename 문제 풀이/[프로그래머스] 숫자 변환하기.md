# [프로그래머스] 숫자 변환하기

## 풀이

DFS로 탐색하며 visited[y]를 반환한다.

```js
const solution = (x, y, n) => {
  if (x === y) return 0;

  const stack = [[x, 0]];
  const visited = Array(y + 1).fill(Infinity);

  const convert = (next, count) => {
    if (next > y || visited[next] <= count + 1) return;
    stack.push([next, count + 1]);
    visited[next] = count + 1;
  };

  while (stack.length) {
    const [curr, count] = stack.pop();
    convert(curr * 3, count);
    convert(curr * 2, count);
    convert(curr + n, count);
  }

  return visited[y] === Infinity ? -1 : visited[y];
};
```
